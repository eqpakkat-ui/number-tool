<script type="module">
import { initializeApp } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-app.js";
import { getDatabase, ref, push, onValue, remove } from "https://www.gstatic.com/firebasejs/10.7.1/firebase-database.js";

const firebaseConfig = {
  apiKey: "AIzaSyA....",
  authDomain: "number-tool-ab240.firebaseapp.com",
  databaseURL: "https://number-tool-ab240-default-rtdb.asia-southeast1.firebasedatabase.app",
  projectId: "number-tool-ab240"
};

const app = initializeApp(firebaseConfig);
const db = getDatabase(app);

let allNumbers = [];
let duplicateCount = 0;

// 🔥 sync realtime
onValue(ref(db, "numbers"), (snapshot) => {
    const data = snapshot.val();

    if (!data) {
        allNumbers = [];
        updateUI();
        return;
    }

    allNumbers = Object.values(data);
    updateUI();
});

// 🔫 ยิงบาร์โค้ด
document.getElementById("input").addEventListener("keydown", function(e){

    if(e.key === "Enter"){
        e.preventDefault();

        let value = this.value.trim();
        if(!value) return;

        // ❌ ไม่ใช่ตัวเลข
        if(!/^[0-9]+$/.test(value)){
            showAlert("❌ ไม่ใช่ตัวเลข", true);
            this.value = "";
            return;
        }

        // 🔥 เช็คซ้ำก่อนยิง
        if(allNumbers.includes(value)){
            showAlert("❌ เลขซ้ำ: " + value, true);
            this.value = "";
            return;
        }

        // ✅ เพิ่มข้อมูล
        push(ref(db, "numbers"), value);

        this.value = "";
    }
});

// 🔥 อัปเดต UI
function updateUI(){

    const output = document.getElementById("output");
    const stats = document.getElementById("stats");

    let seen = new Set();
    let duplicates = [];

    allNumbers.forEach(n=>{
        if(seen.has(n)) duplicates.push(n);
        else seen.add(n);
    });

    duplicateCount = duplicates.length;

    stats.innerHTML = `จำนวน: ${allNumbers.length} | ซ้ำ: ${duplicateCount}`;

    // ⭐ ตัวล่าสุดจริง (ตัวท้าย)
    const lastIndex = allNumbers.length - 1;

    let display = allNumbers.map((n, index) => {

        // 🟡 ตัวล่าสุดเท่านั้น
        if(index === lastIndex){
            return `<span class="highlight">${n}</span>`;
        }

        // 🔴 ตัวซ้ำ
        if(duplicates.includes(n)){
            return `<span class="duplicate">${n}</span>`;
        }

        return n;
    });

    output.innerHTML = display.join(",");
}

// 📋 Copy
window.copyResult = function(){
    if(!allNumbers.length) return;
    navigator.clipboard.writeText(allNumbers.join(","));
    showAlert("✅ คัดลอกแล้ว");
}

// ♻️ Reset
window.clearData = function(){
    remove(ref(db, "numbers"));
    showAlert("🧹 ล้างข้อมูลแล้ว");
}

// 🔔 แจ้งเตือน
function showAlert(msg, isError=false){

    const alertBox = document.getElementById("alert");

    alertBox.innerHTML = `
        <div style="
            background:${isError ? '#fee2e2' : '#dcfce7'};
            color:${isError ? '#b91c1c' : '#166534'};
            padding:15px;
            border-radius:12px;
        ">
        ${msg}
        </div>
    `;

    setTimeout(()=> alertBox.innerHTML="", 3000);
}
